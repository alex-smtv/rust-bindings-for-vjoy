/**
    Describes a button state of a vJoy device.
*/
// We cannot use bool, so we simulate it
#[repr(u8)]
#[derive(Debug, Copy, Clone, PartialEq, Eq, Hash)]
pub enum VJDButtonState {
    Pressed = 1,
    Released = 0,
}
